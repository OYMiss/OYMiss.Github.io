安装步骤

# 安装
mkdir hexo && cd hexo
npm install hexo-cli -g
npm install hexo --save

# 初始化
mkdir oymiss.github.io && hexo init
# git clone https://github.com/OYMiss/oymiss.github.io.git && cd oymiss.github.io
cd oymiss.github.io
npm install

# 先安装 pandoc
# 自带引擎会对 `\\` 进行转义，需要更换引擎。
npm uninstall hexo-renderer-marked --save
npm install hexo-renderer-pandoc --save

# 更改 /nunjucks/src/lexer.js 文件
hexo clean && hexo s

# 发布
npm install hexo-deployer-git --save
hexo d
